# NN training

n <- names(train)
f <- as.formula(paste("class ~", paste(n[!n %in% "class"], collapse = " + ")))
#tr <- train

run_once <- function(tr, val, weights = NULL) {
    nn_mod <- neuralnet(f,data=tr,
                        hidden=c(64, 32, 8),
                        threshold = Inf,
                        stepmax = 1,
                        rep = 1,
                        startweights = NULL,
                        #act.fct= "relu",
                        err.fct = "ce",
                        lifesign = "minimal",
                        linear.output=F)
    
    loss_tr <- nn_mod[["result.matrix"]][[1,1]]
    
    val_y <- predict(nn_mod, newdata = val)
    #val_y <-round(val_y)
    loss_val <- mean(val_y * (1 - val[["class"]]) - val[["class"]] * (1 - val_y)) #not sure it's ok
    #loss_val <- loss.cross.entropy(val_y, val)
    
    list(
        loss_tr = loss_tr,
        loss_val = loss_val,
        weights = nn_mod[["weights"]]
    )
}

run_n <- function(tr, val, m, weights = NULL) {
    perf_df <- data.frame(
        epoch = seq_len(m),
        loss_tr = NA_real_,
        loss_val = NA_real_
    )
    
    first <- run_once(tr, val, weights = weights)
    perf_df[1, 2:3] <- c(first[["loss_tr"]], first[["loss_val"]])
    
    epoch <- first
    for (i in seq_len(m - 1)) {
        epoch[] <- run_once(tr, val, weights = epoch[["weights"]])
        perf_df[i, 2:3] <- c(epoch[["loss_tr"]], epoch[["loss_val"]])
    }
    
    list(
        perf_df = perf_df,
        last_weights_set = epoch[["weights"]]
    )
}
